Simulação de Robô Montador de Hambúrguer - C#

Código em C#

using System; class RoboHamburguer { public void PrepararHamburguer() { Console.WriteLine("Iniciando a preparação do hambúrguer..."); AdicionarPao(); AdicionarCarne(); AdicionarQueijo(); AdicionarVerduras(); FecharHamburguer(); } private void AdicionarPao() { Console.WriteLine("Adicionando pão..."); } private void AdicionarCarne() { Console.WriteLine("Adicionando carne..."); } private void AdicionarQueijo() { Console.WriteLine("Adicionando queijo..."); } private void AdicionarVerduras() { Console.WriteLine("Adicionando alface e tomate..."); } private void FecharHamburguer() { Console.WriteLine("Fechando o hambúrguer..."); } } class Program { static void Main() { RoboHamburguer robo = new RoboHamburguer(); robo.PrepararHamburguer(); } }
